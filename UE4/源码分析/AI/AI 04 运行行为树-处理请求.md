```
void UBehaviorTreeComponent::TickComponent(float DeltaTime, enum ELevelTick TickType, FActorComponentTickFunction *ThisTickFunction)
{
    if (bRequestedFlowUpdate)
    {
        ProcessExecutionRequest();
    }
}
```
